// Arduino Reverse Parking Alert System
// Uses an ultrasonic sensor to detect obstacles while reversing
// and sounds a buzzer when an object is too close.

const int trigPin = 3;    // Trigger pin of ultrasonic sensor
const int echoPin = 2;    // Echo pin of ultrasonic sensor
const int buzzerPin = 6;  // Buzzer pin

long duration;
int distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(buzzerPin, OUTPUT);
  Serial.begin(9600); // Initialize serial communication
}

void loop() {
  // Trigger the ultrasonic sensor
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  // Read the echo time
  duration = pulseIn(echoPin, HIGH);

  // Calculate distance in cm
  distance = duration * 0.034 / 2;

  // Print distance to Serial Monitor
  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  // Sound buzzer if object is closer than 30 cm
  if (distance < 30 && distance > 0) { // added distance > 0 to avoid false triggers
    tone(buzzerPin, 1000); // Alert sound at 1kHz
  } else {
    noTone(buzzerPin); // Turn off buzzer
  }

  delay(500); // Half-second delay between measurements
}
